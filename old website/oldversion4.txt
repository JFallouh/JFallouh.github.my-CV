        // Reusable function to focus on an element
        function focusElement(element) {
            // Remove focus from all other elements
            document.querySelectorAll('.focused-section').forEach(item => {
                item.classList.remove('focused-section');
            });

            // Highlight the selected element
            element.classList.add('focused-section');

            // Listen for clicks outside the element or pressing the Escape key to remove focus
            document.addEventListener('click', clearFocusOnClickOutside);
            document.addEventListener('keydown', clearFocusOnEscape);
        }

        // Function to remove focus
        function clearFocus() {
            document.querySelectorAll('.focused-section').forEach(item => {
                item.classList.remove('focused-section');
            });

            // Remove the event listeners to avoid unnecessary calls
            document.removeEventListener('click', clearFocusOnClickOutside);
            document.removeEventListener('keydown', clearFocusOnEscape);
        }

        
        // Remove focus when clicking outside the focused element, but not when clicking on an h5 that triggers focus
function clearFocusOnClickOutside(event) {
    const focusedElement = document.querySelector('.focused-section');
    const h5Elements = document.querySelectorAll('.carousel-caption h5');

    // Check if the click is outside the focused element and not on an h5 element
    if (focusedElement && !focusedElement.contains(event.target) && !Array.from(h5Elements).includes(event.target)) {
        clearFocus();
    }
}


        // Remove focus when pressing the Escape key
        function clearFocusOnEscape(event) {
            if (event.key === 'Escape') {
                clearFocus();
            }
        }

        document.querySelectorAll('.carousel-caption h5').forEach(h5 => {
            h5.addEventListener('click', function () {
                const projectTitle = this.textContent.trim();

                // Find the corresponding project card by matching the title
                const projectCard = Array.from(document.querySelectorAll('.project-card')).find(card =>
                    card.querySelector('h3').textContent.trim() === projectTitle
                );

                if (projectCard) {
                    // Scroll the project card into view and focus it
                    projectCard.scrollIntoView({ behavior: 'smooth' });
                    focusElement(projectCard);
                }
            });
        });

        // Add onclick event to all project cards for direct click focusing
        document.querySelectorAll('.project-card').forEach(card => {
            card.addEventListener('click', function () {
                focusElement(this);
            });
        });
